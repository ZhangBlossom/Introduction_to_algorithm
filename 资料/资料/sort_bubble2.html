public static void bubble(int[] a) {
  final int n = a.length - 1;
  for (int j = 0; j < n; j++) {
    boolean swapped = false;
    for (int i = 0; i < n - j; i++) {
      if (a[i] > a[i + 1]) {
        swap(a, i, i + 1);
        swapped = true;
      }
    }
    if (!swapped) {
      break;
    }
  }
}
