public static int binarySearchLeftmost(int[] a, int target) {
  int i = 0, j = a.length - 1;
  while (i <= j) {
      int m = (i + j) >>> 1;
      if (target <= a[m]) {       
          j = m - 1;              
      } else {                    
          i = m + 1;
      }
  }
  return i;
}
